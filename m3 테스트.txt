프로그램 실행 결과: 

좋았던 점 : 별다른 모듈 없이 실행되어 코드가 가벼워서 좋았다. 대체운동 추천과 난이도에 따른 세트 수 조정 등의 기능도
들어가 있어서 요구사항도 어느정도 반영되었다.

아쉬웠던 점 : 카메라를 연결 할 수 없었기에 무언가를 이미지로 검사하는 요구사항을 만족할 수 없었다. 대체 운동의 수가
다양하지 않았다.

배운 점 : 어느정도 프롬프트로 입력하여 프로그램의 뼈대는 만들 수 있는 것을 깨달았다. 하지만 구체적이지 않거나 내용에
오류가 있는 경우도 있었기에 더욱 비판적으로 검수해야 한다는 것이 조금 신경쓰였다. 다음에 다시 만들게 된다면, 요구사항과
유스케이스 등을 더욱 구체화시켜서 원하는 결과에 더 근접할 수 있도록 하고싶다.

function analyzePosture(){

    const states = [
        {
            posture:"올바른 자세",
            feedback:"현재 자세가 좋습니다.",
            warning:"관절 부담 없음"
        },
        {
            posture:"무릎 과굴곡",
            feedback:"무릎 각도를 줄이세요.",
            warning:"무릎 부상 위험"
        },
        {
            posture:"허리 굽음",
            feedback:"허리를 곧게 펴세요.",
            warning:"허리 부담 증가"
        },
        {
            posture:"어깨 불균형",
            feedback:"양쪽 어깨 높이를 맞추세요.",
            warning:"어깨 관절 위험"
        }
    ];

    let result =
        states[Math.floor(Math.random()*states.length)];

    document.getElementById("posture").innerHTML =
        "자세 상태 : " + result.posture;

    document.getElementById("feedback").innerHTML =
        "피드백 : " + result.feedback;

    document.getElementById("joint").innerHTML =
        "관절 경고 : " + result.warning;
}

function recommendAlternative(){

    let ex =
        document.getElementById("exercise").value;

    let recommendation = "";

    if(ex==="스쿼트")
        recommendation="레그프레스";

    if(ex==="벤치프레스")
        recommendation="푸쉬업";

    if(ex==="데드리프트")
        recommendation="백 익스텐션";

    document.getElementById("alternative").innerHTML =
        "추천 운동 : " + recommendation;
}

function recommendIntensity(){

    let level =
        document.getElementById("level").value;

    let result="";

    if(level==="초급")
        result="3세트 × 10회";

    if(level==="중급")
        result="4세트 × 12회";

    if(level==="고급")
        result="5세트 × 15회";

    document.getElementById("intensity").innerHTML =
        "추천 강도 : " + result;
}
